GLOBAL.localStorage = {
  storage: {},
  getItem: function (key) {
    return this.storage[key];
  },
  setItem: function (key, item) {
    if (typeof item !== 'string') {
      item = JSON.stringify(item);
    }
    this.storage[key] = item;
  },
  clear: function () {
    this.storage = {};
  }
};
